import pandas as pd
import numpy as np
from sklearn.cluster import KMeans
dataset = pd.read_csv(r'D:\1work\0 Postdoc\0_Europe_groundwater\GISdata\prediction\traindata1.csv')
X = dataset.drop(['Risk'], axis=1)
X.fillna(0, inplace=True)
y = dataset['Risk']
locations = dataset[['Longitude', 'Latitude']]
feature_names = X.columns.tolist()
n_clusters = 24  
kmeans = KMeans(n_clusters=n_clusters, random_state=4564)
clusters = kmeans.fit_predict(locations)
dataset['cluster'] = clusters  
cluster_stats = dataset.groupby('cluster')['Risk'].agg(['count', 'sum'])
pos_clusters = cluster_stats[cluster_stats['sum'] > 0].index.tolist()
non_pos_clusters = cluster_stats[cluster_stats['sum'] == 0].index.tolist()
np.random.seed(1432)
np.random.shuffle(pos_clusters)
np.random.shuffle(non_pos_clusters)
n_train = int(n_clusters * 0.7)
n_test = n_clusters - n_train
n_pos_train = int(len(pos_clusters) * 0.7)  
n_pos_test = len(pos_clusters) - n_pos_train
train_pos_clusters = pos_clusters[:n_pos_train]
test_pos_clusters = pos_clusters[n_pos_train:]

n_non_train = n_train - len(train_pos_clusters)
n_non_test = n_test - len(test_pos_clusters)
train_non_pos_clusters = non_pos_clusters[:n_non_train]
test_non_pos_clusters = non_pos_clusters[n_non_train : n_non_train + n_non_test]

train_clusters = train_pos_clusters + train_non_pos_clusters
test_clusters = test_pos_clusters + test_non_pos_clusters

train_mask = dataset['cluster'].isin(train_clusters)
test_mask = dataset['cluster'].isin(test_clusters)
X_train = X[train_mask]
y_train = y[train_mask]
X_test = X[test_mask]
y_test = y[test_mask]

import numpy as np
import xgboost as xgb
from sklearn.model_selection import KFold
from sklearn.metrics import roc_auc_score, accuracy_score, recall_score, confusion_matrix
from sklearn.model_selection import cross_val_score, StratifiedKFold

kf = KFold(n_splits=10, shuffle=True, random_state=111)
cv_auc, cv_acc, cv_recall, cv_specificity = [], [], [], []

for fold, (train_idx, val_idx) in enumerate(kf.split(X_train_balanced)):
    X_tr, X_val = X_train_balanced[train_idx], X_train_balanced[val_idx]
    y_tr, y_val = y_train_balanced[train_idx], y_train_balanced[val_idx]
    model = xgb.XGBClassifier(use_label_encoder=False, eval_metric='logloss', random_state=fold)
    model.fit(X_tr, y_tr)    
    val_pred_prob = model.predict_proba(X_val)[:, 1]
    val_pred = (val_pred_prob > 0.5).astype(int)

    auc = roc_auc_score(y_val, val_pred_prob)
    acc = accuracy_score(y_val, val_pred)
    recall = recall_score(y_val, val_pred)  # sensitivity
    tn, fp, fn, tp = confusion_matrix(y_val, val_pred).ravel()
    specificity = tn / (tn + fp)
    cv_auc.append(auc)
    cv_acc.append(acc)
    cv_recall.append(recall)
    cv_specificity.append(specificity)
    print(f"Fold {fold+1}: AUC={auc:.3f}, ACC={acc:.3f}, Sensitivity={recall:.3f}, Specificity={specificity:.3f}")
print("AUC:         %.4f" % np.mean(cv_auc))
print("Accuracy:    %.4f" % np.mean(cv_acc))
print("Sensitivity: %.4f" % np.mean(cv_recall))
print("Specificity: %.4f" % np.mean(cv_specificity))

import sqlite3
import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler
import xgboost as xgb

conn = sqlite3.connect(r'D:\1work\0 Postdoc\0_Europe_groundwater\GISdata\prediction\ML_output.db')
query = "SELECT * FROM ML_data"
data_pred = pd.read_sql_query(query, conn)
conn.close()
features_pred = data_pred.iloc[:, 1:33]

X_pred_scaled = scaler.transform(features_pred)

n_iterations = 100
predictions = np.zeros((n_iterations, len(X_pred_scaled)), dtype=np.float32)

for i in range(n_iterations):

    indices = np.random.choice(len(X_train_balanced), len(y_train_balanced), replace=True)
    X_resample = X_train_balanced[indices]
    y_resample = y_train_balanced[indices]

    model = xgb.XGBClassifier(use_label_encoder=False,eval_metric='logloss',random_state=i,
                              learning_rate=0.001, n_estimators=900,max_depth=7,gamma=0,subsample=0.5,
                              colsample_bytree=1,min_child_weight=11,reg_alpha=4,reg_lambda=2)
    model.fit(X_resample, y_resample)


    preds = model.predict_proba(X_pred_scaled)[:, 1]
    predictions[i, :] = preds


mean_predictions = np.mean(predictions, axis=0)
std_predictions = np.std(predictions, axis=0)

for i in range(len(mean_predictions)):
    print(f"Sample {i + 1} - Mean Prediction: {mean_predictions[i]:.4f}, Std Dev: {std_predictions[i]:.4f}")

results = pd.DataFrame({
    'id': np.arange(1, len(mean_predictions) + 1),
    'Mean_pred': mean_predictions,
    'Std_Dev': std_predictions,
})


output_db_path = r'D:\1work\0 Postdoc\0_Europe_groundwater\GISdata\prediction\Std_Dev.db'
conn = sqlite3.connect(output_db_path)
results.to_sql('Std_Dev', conn, if_exists='replace', index=False)
conn.close()
